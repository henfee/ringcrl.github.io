JavaScript 语句

发表于 2016-04-21   |  

语句

block

块语句常用于组合0 ~ 多个语句。块语句用一对花括号定义。
请注意：没有块级作用域
实例1

for (var i = 0; i < 10; i++) {
var str = "hi";
console.log(str);
}
=
var i = 0
for (; i < 10; i++) {
    var str = "hi";
    console.log(str);
}

实例2

{
    var x = 1;
    // .....
}
=
var x = 1;
{
    // .....
}

实例3

function foo() {
    var a = 1;
    console.log(a); // 1
}
foo();
console.log(typeof a); // undefined

try catch

有两种情况
一种是try catch finally,做某件事 捕获错误 结果
另一种是try finally,做某件事 不管有没有错误 直接出结果

function

函数声明(会被函数前置)

fd(); // true

function fd() {
    // do sth.
    return true;
}

函数表达式

fe(); // TypeError

var fe = function() {
    // do sth.
};

for…in

var p;
var obj = {x : 1, y: 2}
for (p in obj) {
}
  1. 顺序不确定
  2. enumerable为false时不会出现
  3. for in对象属性时受原型链影响

switch

var val = 2;

switch(val) {
    case 1:
        console.log(1);
        break;
    case 2:
        console.log(2);
        break;
    default:
        console.log(0);
        break;
}

结果是:2

switch(val) {
    case 1:
        console.log(1);
    case 2:
        console.log(2);
    default:
        console.log(0);
}

结果是:2,0

switch(val) {
    case 1:
    case 2:
    case 3:
        console.log(123);
        break;
    case 4:
    case 5:
        console.log(45); 
        break;
    default:
        console.log(0);
}

结果是:123

严格模式

严格模式是一种特殊的执行模式，它修复了部分语言上的不足，提供更强的错误检查，并增强安全性。
'use strict';

不允许用with

结果:SyntaxError

不允许未声明的变量被赋值

!function() {
    'use strict';
     x = 1;
      console.log(window.x);
}();

结果:ReferenceError

delete参数、函数名报错

!function(a) {
    'use strict';
    delete a;
}(1);

结果:SyntaxError

delete不可配置的属性报错

!function(a) {
    'use strict';
    var obj = {};
    Object.defineProperty(obj, 
        'a', {configurable : false});
    delete obj.a;
}(1);

结果:TypeError

对象字面量重复属性名报错

!function() {
    'use strict';
    var obj = {x : 1, x : 2};
}();

结果:SyntaxError

禁止八进制字面量

!function() {
    'use strict';
    console.log(0123);
}();

结果:SyntaxError

eval, arguments变为关键字，不能作为变量、函数名

!function() {
    'use strict';
    function eval(){}
}();

结果:SyntaxError
